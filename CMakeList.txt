cmake_minimum_required(VERSION 3.0.2)
project(franka_data_collector)

## Find catkin macros and libraries
find_package(catkin REQUIRED COMPONENTS
  roscpp
  rospy
  std_msgs
  geometry_msgs
  sensor_msgs
  franka_msgs
  tf
)

## Declare a catkin package
catkin_package(
  CATKIN_DEPENDS roscpp rospy std_msgs geometry_msgs sensor_msgs franka_msgs tf
)

## Include directories
include_directories(
  ${catkin_INCLUDE_DIRS}
)

## Add executables
add_executable(data_collector_node src/data_collector_node.cpp)
target_link_libraries(data_collector_node ${catkin_LIBRARIES})

## Install targets
install(TARGETS data_collector_node
  RUNTIME DESTINATION ${CATKIN_PACKAGE_BIN_DESTINATION}
)

## Install other files (e.g., launch files)
install(DIRECTORY launch/
  DESTINATION ${CATKIN_PACKAGE_SHARE_DESTINATION}/launch
)
